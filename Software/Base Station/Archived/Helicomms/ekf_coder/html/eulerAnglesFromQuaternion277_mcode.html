1function [roll,pitch,yaw] = eulerAnglesFromQuaternion(x)%#codegen
 2    q0 = x(1); 
 3    q1 = x(2);  
 4    q2 = x(3);  
 5    q3 = x(4);   
 6
 7    roll = atan2(2*(q0*q1+q2*q3),1-2*(q1^2+q2^2))*180/pi;
 8    pitch = asin(2*(q0*q2-q3*q1))*180/pi;
 9    yaw = atan2(2*(q0*q3+q1*q2),1-2*(q2^2+q3^2))*180/pi;    
10end
